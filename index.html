Selection Sort

                    
                        function selectionSort(arr) {
                            for (let i=0; i < arr.length; i++) {
                                let min=i;
                                for (let n=i+1; n < arr.length; n++) {
                                    if (arr[min] > arr[n]) {
                                        min=n;
                                    }
                                }
                                const change = arr[min];
                                arr[min]=arr[i];
                                arr[i]= change;
                            }
                            return arr;
                        }
